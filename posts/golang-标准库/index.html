Package builtin provides documentation for Go's predeclared identifiers.
The items documented here are not actually in package builtin
but their descriptions here allow godoc to present documentation
for the language's special identifiers.

标准库 Builtin 提供Go语言预定义的类型和函数。文件中的注释对各种类型和函数的作用做出了解释。

这些预定义函数在编译过程被执行，这里只是为了我们更好地理解数据类型和函数的作用。

下面是源代码和中文解释：

package builtin

// bool 是一个布尔类型，有true和false，也就是1和0两种值。
type bool bool

// true和false是两种无类型布尔值。
const (
	true  = 0 == 0
	false = 0 != 0
)

// uint8 是所有无符号 8-bit 整数类型的集合。
// 有效值范围：0-255。
type uint8 uint8

// uint16 是所有无符号 16-bit 整数类型的集合。
// 有效值范围：0-65535。
type uint16 uint16

// uint32 是所有无符号 32-bit 整数类型的集合。
// 有效值范围：0-4294967295。
type uint32 uint32

// uint64 是所有无符号 64-bit 整数类型的集合。
// 有效值范围：0-18446744073709551615。
type uint64 uint64

// int8 是所有有符号 8-bit 整数类型的集合。
// 有效值范围：-128-127。
type int8 int8

// int16 是所有有符号 16-bit 整数类型的集合。
// 有效值范围：-32768-32767。
type int16 int16

// int32 是所有有符号 32-bit 整数类型的集合。
// 有效值范围：-2147483648-2147483647。
type int32 int32

// int64 是所有有符号 64-bit 整数类型的集合。
// 有效值范围：-9223372036854775808-9223372036854775807。
type int64 int64

// float32 是所有 32-bit 浮点数的集合，规则遵循[IEEE-754](https://ieeexplore.ieee.org/document/8766229)标准。
type float32 float32

// float64 是所有 64-bit 浮点数的集合，规则遵循[IEEE-754](https://ieeexplore.ieee.org/document/8766229)标准。
type float64 float64

// complex64 是所有 float32 复数（实部和虚部）的集合。
type complex64 complex64

// complex128 是所有 float64 复数（实部和虚部）的集合。
type complex128 complex128

// string 是字符串，是所有 8-bit 字节的集合，但并一定符合传统 UTF-8 编码文本。
// 字符串可以为空（”“），但不能是 nil，字符串的值是不可变的。
type string string

// int 是一个整数类型，大小为32bit，与int32类似。
type int int

// uint 是无符号整数类型，大小为32bit，与uint32类似。
type uint uint

// uintptr 是无符号整数指针类型。
type uintptr uintptr

// byte作为uint8数据类型的别名，用于处理基本的输出输出和文本数据。
type byte = uint8

// rune作为int32数据类型的别名，用于处理字符值。
type rune = int32

// iota 作为无符号整数类型，主要用来做常量的索引，方便数据遍历和循环。
const iota = 0

// nil 类型用来作指针（pointer）、channel（信道）、func（函数）、interface（接口）、map和slice等数据类型的零值。
var nil Type

type Type int

type Type1 int

type IntegerType int

type FloatType float32

type ComplexType complex64

// append 函数用于添加元素到切片（slice）的结尾。
// 添加过程中如何有足够的容量（capacity），那么就直接添加。
// 否则在底层array中分配新的空间，以支持添加操作完成。
// 整个操作最终会返回新的切片，因此有必要将结果值传给变量。
// 例如下面的操作：
// slice = append(slice, elem1, elem2)
// slice = append(slice, anotherSlice...)
// 当然，append 也能将字符串类型加入到字节类型切片后面：
// slice = append([]byte("hello "), "world"...)
func append(slice []Type, elems ...Type) []Type

// copy 函数用于将原数据（source）切片拷贝到目标数据（destination）切片。
// 拷贝过程中可能发生重叠。
// 函数返回值是原数据长度（len(src)）或目标数据长度（len(dst)）拷贝数量。
func copy(dst, src []Type) int

// delete 用于删除键-值对中的数据。如果本身为空或没有对应的键值，删除操作将无事发生（no-op）。
func delete(m map[Type]Type1, key Type)

// len 函数返回传入值的长度，根据传入的类型，有如下几种：
// Array：数组中的元素数量。
// Pointer to array：指针数组中的元素数量（*v）。
// Slice，Map：切片和键-值对的元素数量。
// String：字符串 bytes 的数量。
// Channel：信道缓冲（channel buffer）中的队列（未读）数量。
func len(v Type) int

// cap 函数返回传入类型的容量。如：
// Array：数组中的元素数量。
// Pointer to array：指针数组中的元素数量（*v）。
// Slice：切片可达到的最大长度。
// Channel：信道缓冲（channe buffer）的容量。
func cap(v Type) int

// make 函数用来初始化和创建一个数据类型的对象，只支持slice、map和chan。
// 数据类型和类型大小作为参数传递。返回的结果与传入的类型一致。
// Slice：make([]int, 0, 10) 第一个参数是类型，第二个是长度，第三个是容量。容量可以表示底层的数组，不能比长度小。
// Map：make(map([int]string), 10) 第一个参数是类型，第二个是大小。
// Channel：make(chan bool, 10) 第一个是类型，第二个是buffer容量，如果为零，则为无缓冲的信道。
func make(t Type, size ...IntegerType) Type

// new 函数用于为数据类型分配内存。第一个参数传入类型，不是值。返回的结果是一个零值的指针类型。
func new(Type) *Type

// complex 函数用于将传入的实部和虚部作为复数返回，支持float32和float64数据类型。
func complex(r, i FloatType) ComplexType

// real 函数将传入的复数实部作为浮点数返回。
func real(c ComplexType) FloatType

// imag 函数将传入的复数虚部作为浮点数返回。
func imag(c ComplexType) FloatType

// close 函数用于关闭信道，此时传入的参数类型（信道）必须是双向的或者发送状态。
// 关闭后接收逻辑：
// x, ok := <-c
// 因为信道已被关闭，此时接收的ok值一定是false。
func close(c chan<- Type)

// panic 函数用于立即停止当前正在执行的协程（goroutine），任务结束后是一个不为零的退出代码。
// 当 panic 函数发生时，可以做一次defer函数，以完成必须的任务。
// 通常panic函数会与recover函数一起出现，后者用于捕获可能发生的异常错误。
func panic(v interface{})

// recover 函数用于捕获并发过程中可能出现的异常现象（Panic）。
// 函数返回空时，表示无异常，否则出现异常且已被捕获，需要进一步处理。
func recover() interface{}

// print 函数用于打印东西，例如错误，值等。
// 函数也可用于自助抽样法（bootstrapping） 和代码调试。
func print(args ...Type)

// println 函数与print类似，都是打印，到println能打印的同时换行。
func println(args ...Type)

// error 接口是基本的错误处理接口，方便其他数据结构和操作实现。
type error interface {
	Error() string
}
